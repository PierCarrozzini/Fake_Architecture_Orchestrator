# main.tf
terraform {
  required_providers {
    docker = {
      source  = "kreuzwerker/docker"
      version = "3.0.2"
    }
  }
}

provider "docker" {
  host    = "tcp://localhost:2375/"
}

# Docker network
resource "docker_network" "example_network" {
  name = "example_network"
}

# Nginx container
resource "docker_container" "example_nginx" {
  name  = "example_nginx"
  image = "nginx:alpine"
  ports {
    internal = 80
    external = 8080
  }
  networks_advanced {
    name = docker_network.example_network.name
  }
  depends_on = [docker_network.example_network]
}
